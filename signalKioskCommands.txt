
#Commands transmitted into tcp-command host/port=88888

Set SignalK server host: SIGNALKHOST = "10.10.10.1"   (this is where telemtry is retreived for display)

Set SignalK server port: SIGNALKPORT = "20311"

Set SignalKiosk display brightness: BRIGHT = "1-15"

Set SignalKiosk refresh rate in seconds:  DISPREFRESH = "0.5,1.0,2.0"

Set SignalKiosk telemetry label OFF/ON: TELEMLABEL = "ON"

Set SignalKiosk telemetry label appearance rate: LABELRATE = "integer-seconds" i.e., 10 seconds, 15 sec, 30sec.
